Expenses.vb (Validated Data Entry)


' Expenses.vb - Handles validated insertion of expense data
Imports DocumentFormat.OpenXml.Packaging
Imports DocumentFormat.OpenXml.Spreadsheet
Imports System.IO

Module Expenses
    Private filePath As String = "ExpensesTracker.xlsx"

    Sub InsertExpenseEntry()
        Try
            If Not File.Exists(filePath) Then
                Console.WriteLine("Excel file not found.")
                Return
            End If

            Using document As SpreadsheetDocument = SpreadsheetDocument.Open(filePath, True)
                Dim workbookPart = document.WorkbookPart
                Dim sheet = workbookPart.Workbook.Descendants(Of Sheet)().
                            FirstOrDefault(Function(s) s.Name = "Expenses")
                If sheet Is Nothing Then
                    Console.WriteLine("Expenses sheet not found.")
                    Return
                End If

                Dim worksheetPart = CType(workbookPart.GetPartById(sheet.Id), WorksheetPart)
                Dim sheetData = worksheetPart.Worksheet.GetFirstChild(Of SheetData)()
                Dim nextId As Integer = GetSafeNextId(sheetData, workbookPart)

                ' Collect inputs from user
                Console.Write("Date (yyyy-MM-dd): ")
                Dim dateSpent = Console.ReadLine()
                Console.Write("Amount Spent: ")
                Dim amountSpent = Console.ReadLine()
                Console.Write("Spent On: ")
                Dim spentOn = Console.ReadLine()
                Console.Write("Category ID: ")
                Dim categoryId = Console.ReadLine()
                Console.Write("Place ID: ")
                Dim placeId = Console.ReadLine()
                Console.Write("Payment Method ID: ")
                Dim paymentMethodId = Console.ReadLine()

                ' Validate foreign keys
                If Not IdExists("Categories", categoryId, workbookPart) Then
                    Console.WriteLine("‚ùå Category ID does not exist.")
                    Exit Sub
                End If
                If Not IdExists("Places", placeId, workbookPart) Then
                    Console.WriteLine("‚ùå Place ID does not exist.")
                    Exit Sub
                End If
                If Not IdExists("PaymentMethods", paymentMethodId, workbookPart) Then
                    Console.WriteLine("‚ùå Payment Method ID does not exist.")
                    Exit Sub
                End If

                ' Add new expense row
                Dim newRow As New Row()
                newRow.Append(CreateTextCell("A", nextId.ToString()))
                newRow.Append(CreateTextCell("B", dateSpent))
                newRow.Append(CreateTextCell("C", amountSpent))
                newRow.Append(CreateTextCell("D", spentOn))
                newRow.Append(CreateTextCell("E", categoryId))
                newRow.Append(CreateTextCell("F", placeId))
                newRow.Append(CreateTextCell("G", paymentMethodId))

                sheetData.AppendChild(newRow)
                worksheetPart.Worksheet.Save()
                Console.WriteLine("‚úÖ Expense entry added successfully.")
            End Using
        Catch ex As Exception
            Console.WriteLine($"An error occurred: {ex.Message}")
        End Try

        Console.WriteLine("Press Enter to return to menu.")
        Console.ReadLine()
    End Sub

    Private Function GetSafeNextId(sheetData As SheetData, workbookPart As WorkbookPart) As Integer
        Dim numericIds = sheetData.Elements(Of Row)().
            Select(Function(r) GetCellValue(workbookPart, r.Elements(Of Cell).FirstOrDefault())).
            Where(Function(value) Integer.TryParse(value, Nothing)).
            Select(Function(value) Integer.Parse(value))
        Return If(numericIds.Any(), numericIds.Max() + 1, 1)
    End Function

    Private Function IdExists(sheetName As String, idToFind As String, workbookPart As WorkbookPart) As Boolean
        Dim sheet = workbookPart.Workbook.Descendants(Of Sheet)().
            FirstOrDefault(Function(s) s.Name = sheetName)
        If sheet Is Nothing Then Return False

        Dim worksheetPart = CType(workbookPart.GetPartById(sheet.Id), WorksheetPart)
        Dim sheetData = worksheetPart.Worksheet.GetFirstChild(Of SheetData)()
        Return sheetData.Elements(Of Row)().
            Any(Function(r) GetCellValue(workbookPart, r.Elements(Of Cell).First()).Trim() = idToFind)
    End Function

    Private Function CreateTextCell(column As String, value As String) As Cell
        Return New Cell With {
            .CellReference = column,
            .DataType = CellValues.String,
            .CellValue = New CellValue(value)
        }
    End Function

    Private Function GetCellValue(workbookPart As WorkbookPart, cell As Cell) As String
        If cell Is Nothing Then Return ""
        If cell.DataType IsNot Nothing AndAlso cell.DataType.Value = CellValues.SharedString Then
            Dim sstPart = workbookPart.GetPartsOfType(Of SharedStringTablePart)().FirstOrDefault()
            Return If(sstPart IsNot Nothing,
                      sstPart.SharedStringTable.ElementAt(Integer.Parse(cell.CellValue.Text)).InnerText,
                      "")
        Else
            Return cell.CellValue?.Text
        End If
    End Function
End Module

Q3. Why does the app ask for IDs instead of names for Categories or Places?

Using numeric IDs mirrors how foreign keys work in relational databases. It guarantees consistency and allows efficient cross-referencing between sheets.

Q4. What is OpenXML and why is it used here?

OpenXML is a .NET SDK that lets you read, write, and manipulate Excel files without needing Microsoft Office installed. It‚Äôs fast, lightweight, and ideal for structured automation like this app.

Q5. How does the app generate a new ID for each expense?

The GetSafeNextId function scans the existing 'Expenses' sheet, finds the highest ID, and returns the next available one. This ensures all entries have unique identifiers without overlap.
